[docs]def rotate_and_flip(coord_map, rotation, flip):
    rotation = (-rotation % 360) // 90
    for i in range(rotation):
        coord_map = list(zip(*coord_map[::-1]))

    if flip:
        coord_map = coord_map[::-1]

    return coord_map
